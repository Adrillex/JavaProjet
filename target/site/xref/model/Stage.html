24  public class Stage {
25  	
26  	private int numStage;
27  	private Bonus bPurse;
28  	public static ArrayList <CrystalBall> cBallList;
29  	public static ArrayList <Purse> purseList;
30  	public static ArrayList <Gate> gateList;
31  	public static ArrayList <Hero> rLorannList;
32  	public static ArrayList <IElement> elements;
33  	public static ArrayList <Monster> monsterList;
34  	
35  	public static Fireball fireball;

37  	public Stage(int numStage, Bonus bPurse){
38  		this.numStage = numStage;
39  		this.bPurse = bPurse;
40  	}
41  
42  	/** the method LoadStage
43  	 * 
44  	 * @return IElements
45  	 * Create ArrayLists of elements for a later user.
46  	 * @throws Exception
47  	 * 	Constructors of elements
48  	 */
49  	public ArrayList <IElement> LoadStage()throws Exception{
50  		cBallList = new ArrayList <CrystalBall>();
51  		purseList = new ArrayList <Purse>();
52  		gateList = new ArrayList <Gate>();
53  		rLorannList = new ArrayList <Hero>();
54  		elements = new ArrayList <IElement>();
55  		monsterList = new ArrayList <Monster>();
56  		fireball = new Fireball(Direction.UP, 19, 11, ImageLoader.fireball[0]);
57  		fireball.setExisting(false);
58  		elements.add(fireball);
59  		
60  		DBConnection instance = DBConnection.getInstance();
61  		final String sql = "{call seeStage" + numStage + "}";
62  		CallableStatement call;
63  		ResultSet rs;
64  		try {
65  			call = instance.getConnection().prepareCall(sql);
66  			call.execute();
67  			rs = call.getResultSet();
68  			
69  		} catch (SQLException e) {
70  			e.printStackTrace();
71  			return elements;
72  		}
73  		try {
74  			while (rs.next()){
75  				switch (rs.getInt("id_elements")) {
76  				case 1:
77  					// horizontal wall
78  					elements.add(new Wall(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.horizontal_bone));
79  					break;
80  				case 2:
81  					// vertical wall
82  					elements.add(new Wall(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.vertical_bone));
83  
84  					break;
85  				case 3:
86  					// round wall
87  					elements.add(new Wall(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.bone));
88  					break;
89  				case 4:
90  					// round wall
91  					CrystalBall crystalBall = new CrystalBall(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.crystal_ball);
92  					cBallList.add(crystalBall);
93  					elements.add(crystalBall);
94  					break;
95  				case 5:
96  					// round wall
97  					Purse purse = new Purse(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.purse, bPurse);
98  					purseList.add(purse);
99  					elements.add(purse);
100 					break;
101 				case 6:
102 					// round wall
103 					Gate gate = new Gate(rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.gate);
104 					gateList.add(gate);
105 					elements.add(gate);
106 					break;
107 				case 7:
108 					break;
109 				case 8:
110 					// round wall
111 					Arrbarr arrbarr = new Arrbarr(Direction.RIGHT ,rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.monster_1);
112 					elements.add(arrbarr);
113 					monsterList.add(arrbarr);
114 					break;
115 				case 9:
116 					// round wall
117 					Cargyv cargyv = new Cargyv(Direction.UP ,rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.monster_2);
118 					elements.add(cargyv);
119 					monsterList.add(cargyv);
120 					break;
121 				case 10:
122 					// round wall
123 					Kyrac kyrac = new Kyrac(Direction.UP ,rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.monster_3);
124 					elements.add(kyrac);
125 					monsterList.add(kyrac);
126 					break;
127 				case 11:
128 					// round wall
129 					Maarcg maarcg = new Maarcg(Direction.UP ,rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.monster_4);
130 					elements.add(maarcg);
131 					monsterList.add(maarcg);
132 					break;
133 				case 12:
134 					Hero hero = new Hero(Direction.UP ,rs.getInt("coord_X"), rs.getInt("coord_Y"), ImageLoader.player);
135 					rLorannList.add(hero);
136 					elements.add(hero);
137 					break;
138 				default:
139 					break;
140 				}
141 			} 
142 		} catch (SQLException e) {
143 				e.printStackTrace();
144 		}
145 		return elements;
146 	}
147 }
