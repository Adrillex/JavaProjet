10  import contract.IModel;
11  
12  public class ViewPanelScore extends JPanel{
13  	
14  	/**SerialVersionUID of the panel **/
15  	private static final long serialVersionUID = 1L;
16  	
17  	/** 	Actual score of the player**/
18  	private int score;
19  	
20  	/** The model **/
21  	private IModel model;
22  	
23  	/**The JLabel which show the score**/
24  	private JLabel showScore;
25  	/**The JLabel which shows the number of monster stil alive **/
26  	private JLabel monsterAlive;
27  	/**The JLabel which shows the number of bonus remaining on the level **/
28  	private JLabel bonusRemaining;
29  	
30  	/*
31  	 * the constructor ViewPanelScore
32  	 * @param model
33  	 * 	The model
34  	 * 	Build the PanelScore
35  	 */
36  	public ViewPanelScore(IModel model){	
37  	this.setLayout(new BoxLayout(this, BoxLayout.LINE_AXIS));
38  	this.setBackground(Color.black);
39  	this.model = model;
40  	showScore = new JLabel("Score : " + score);
41  	monsterAlive = new JLabel("Monster alive : " + this.model.getMonsterNumber());
42  	bonusRemaining = new JLabel("Bonus remaining : " + this.model.getBonusNumber());
43  	
44  	showScore.setForeground(Color.orange);
45  	monsterAlive.setForeground(Color.orange);
46  	bonusRemaining.setForeground(Color.orange);
47  	
48  	}
49  	
50  	/** the method updateScore
51  	 * 	Update the information of the panel
52  	 */
53  	public void updateScore(){
54  		this.removeAll();
55  		showScore.setText("Score : " + this.model.getScore());
56  		monsterAlive.setText("Monster(s) alive: " + this.model.getMonsterNumber());
57  		bonusRemaining.setText("Bonus remaining : " + this.model.getBonusNumber());
58  		this.add(showScore);
59  		this.add(Box.createGlue());
60  		this.add(monsterAlive);
61  		this.add(Box.createGlue());
62  		this.add(bonusRemaining);
63  	}
64  	
65  }
